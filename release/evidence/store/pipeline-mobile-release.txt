name: Mobile Release

on:
  workflow_dispatch:
    inputs:
      submit:
        description: "Submit latest successful production build to stores"
        required: false
        default: "false"

jobs:
  eas-build:
    runs-on: ubuntu-latest
    defaults:
      run:
        working-directory: mobile
    env:
      EXPO_TOKEN: ${{ secrets.EXPO_TOKEN }}
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node
        uses: actions/setup-node@v4
        with:
          node-version: "20"
          cache: "npm"
          cache-dependency-path: mobile/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Validate Expo token
        run: |
          if [ -z "$EXPO_TOKEN" ]; then
            echo "EXPO_TOKEN secret is missing"
            exit 1
          fi

      - name: Build iOS and Android (EAS, no wait)
        run: npx eas-cli build --platform all --profile production --non-interactive --no-wait

      - name: Submit to stores
        if: ${{ github.event.inputs.submit == 'true' }}
        run: npx eas-cli submit --platform all --profile production --non-interactive
